class Solution {
    public int mySqrt(int x) {
        int left = 0;
        int right = x;
        int mid = 0;
        
        if (x == 1 || x == 0) return x;
        while (left < right){
            mid = left + (right - left) / 2;
            if (x / mid < mid){
                right = mid;
            } else if (x / mid > mid) {
                left = mid + 1;
            } else {
                return mid;
            }
        }
        return right - 1;
    }
}
